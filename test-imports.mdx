## ThemeProvider Usage

```tsx
import { ThemeProvider, useTheme, Button } from "liquidify-react";

function App() {
  return (
    <ThemeProvider 
      defaultTheme="light" 
      accentPreset="blue"
      persistAccent={true}
    >
      <MyComponent />
    </ThemeProvider>
  );
}

function MyComponent() {
  const { theme, setTheme, accent, setAccent } = useTheme();
  
  return (
    <div>
      <Button variant="filled" tone="accent">
        Current accent: {accent}
      </Button>
    </div>
  );
}
```
