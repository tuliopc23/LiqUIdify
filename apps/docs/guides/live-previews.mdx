```tsx
    </div>
```

```tsx
function FormValidationExample() {
  const [email, setEmail] = React.useState("");
  const [password, setPassword] = React.useState("");
  const [errors, setErrors] = React.useState({});
  
  const validateEmail = (value) => {
    return /^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(value);
  };
  
  const validatePassword = (value) => {
    return value.length >= 8;
  };
  
  React.useEffect(() => {
    const newErrors = {};
    if (email && !validateEmail(email)) {
      newErrors.email = "Invalid email format";
    }
    if (password && !validatePassword(password)) {
      newErrors.password = "Password must be at least 8 characters";
    }
    setErrors(newErrors);
  }, [email, password]);
  
  return (
    <form className="space-y-4 max-w-sm">
      <GlassFormField 
        label="Email" 
        error={errors.email}
        success={email && !errors.email ? "Valid email" : undefined}
      >
        <GlassInput
          type="email"
          value={email}
          onChange={(e) => setEmail(e.target.value)}
          placeholder="your@email.com"
        />
      </GlassFormField>
      
      <GlassFormField 
        label="Password" 
        error={errors.password}
        success={password && !errors.password ? "Strong password" : undefined}
      >
        <GlassInput
          type="password"
          value={password}
          onChange={(e) => setPassword(e.target.value)}
          placeholder="Enter password"
        />
      </GlassFormField>
      
      <GlassButton 
        type="submit" 
        className="w-full"
        disabled={Object.keys(errors).length > 0 || !email || !password}
      >
        Sign Up
      </GlassButton>
    </form>
  );
}
```

## üõ†Ô∏è Implementation Guide

### Setting Up Live Previews

To implement this preview system in your documentation:

1. **Create the Preview Component**

```tsx
// apps/docs/components/Preview.tsx
import React from 'react';
import { GlassCard, GlassButton, GlassIcon } from '@liquidify/react';

interface PreviewProps {
  children: React.ReactNode;
  title?: string;
  description?: string;
}

export function Preview({ children, title, description }: PreviewProps) {
  const [copied, setCopied] = React.useState(false);
  
  const copyCode = async () => {
    // Implementation for copying code
    setCopied(true);
    setTimeout(() => setCopied(false), 2000);
  };
  
  return (
    <GlassCard variant="elevated" className="mb-6">
      {(title || description) && (
        <div className="p-4 border-b border-liquid-border/50">
          {title && <h3 className="font-semibold mb-1">{title}</h3>}
          {description && <p className="text-sm text-liquid-secondary">{description}</p>}
        </div>
      )}
      
      <div className="p-6 bg-gradient-to-br from-liquid-surface/30 to-liquid-surface/10">
        {children}
      </div>
      
      <div className="px-4 py-2 bg-liquid-surface/20 border-t border-liquid-border/50">
        <GlassButton
          variant="ghost"
          size="sm"
          onClick={copyCode}
          className="ml-auto"
        >
          <GlassIcon name={copied ? "check" : "clipboard"} className="mr-2" />
          {copied ? "Copied!" : "Copy Code"}
        </GlassButton>
      </div>
    </GlassCard>
  );
}
```
