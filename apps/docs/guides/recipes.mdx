## Magnetic Hover Button

;

export function MagneticButton() {

    <div className="group relative inline-flex">

## Responsive Card Grid

```tsx
;

export function CardGrid() {
  const cards = Array.from({ length: 6 }, (_, i) => i + 1);
  return (
    <div className="grid gap-6 sm:grid-cols-2 lg:grid-cols-3">
      {cards.map((n) => (
        <GlassCard key={n} variant="elevated" className="p-6">
          <GlassCard.Title>Card {n}</GlassCard.Title>
          <GlassCard.Description>
            Liquid glass surface with soft depth
          </GlassCard.Description>
        </GlassCard>
      ))}
    </div>
  );
}
```

## Controlled Input with Validation

```tsx
;

export function EmailField() {
  const [email, setEmail] = React.useState("");
  const [error, setError] = React.useState<string | null>(null);

  function onSubmit(e) {
    e.preventDefault();
    setError(/^[^@]+@[^@]+$/.test(email) ? null : "Enter a valid email");
  }

  return (
    <form onSubmit={onSubmit} className="space-y-3">
      <GlassFormField label="Email" error={error ?? undefined}>
        <GlassInput
          type="email"
          value={email}
          onChange={(e) => setEmail(e.target.value)}
          placeholder="you@example.com"
        />
      </GlassFormField>
      <GlassButton type="submit">Submit</GlassButton>
    </form>
  );
}
```

## Modal + Focus Trap

```tsx
;

export function ModalExample() {
  const [open, setOpen] = React.useState(false);
  return (
    <>
      <GlassButton onClick={() => setOpen(true)}>Open</GlassButton>
      <GlassModal open={open} onOpenChange={setOpen}>
        <GlassFocusTrap>
          <div className="p-6 space-y-3">
            <h3 className="text-lg font-semibold">Dialog</h3>
            <p>Focus stays within the dialog.</p>
            <GlassButton onClick={() => setOpen(false)}>Close</GlassButton>
          </div>
        </GlassFocusTrap>
      </GlassModal>
    </>
  );
}
```

## Toast on Action

```tsx
;

export function ToastOnSave() {
  const [toasts, setToasts] = React.useState<string[]>([]);
  return (
    <div>
      <GlassButton onClick={() => setToasts((t) => ["Saved!", ...t])}>
        Save
      </GlassButton>
      <div className="fixed bottom-4 right-4 space-y-2">
        {toasts.map((t, i) => (
          <GlassToast key={i}>{t}</GlassToast>
        ))}
      </div>
    </div>
  );
}
```

## Notes

- Use `className` to fine‑tune glass intensity per component.
- Prefer semantic tokens over raw colors.
