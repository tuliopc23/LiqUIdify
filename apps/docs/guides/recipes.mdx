### Enhanced Magnetic Button

```tsx
import { useState, useRef, useCallback } from 'react';
import { GlassButton } from '@liquidify/react';

export function MagneticButton({ 
  children, 
  magneticStrength = 12,
  glowIntensity = 0.3,
  ...props 
}) {
  const buttonRef = useRef<HTMLButtonElement>(null);
  const [position, setPosition] = useState({ x: 0, y: 0 });
  const [isHovered, setIsHovered] = useState(false);

  const handleMouseMove = useCallback((e: React.MouseEvent) => {
    if (!buttonRef.current) return;
    
    const rect = buttonRef.current.getBoundingClientRect();
    const centerX = rect.left + rect.width / 2;
    const centerY = rect.top + rect.height / 2;
    
    const deltaX = e.clientX - centerX;
    const deltaY = e.clientY - centerY;
    
    setPosition({
      x: deltaX * (magneticStrength / 100),
      y: deltaY * (magneticStrength / 100)
    });
  }, [magneticStrength]);

  const handleMouseLeave = useCallback(() => {
    setPosition({ x: 0, y: 0 });
    setIsHovered(false);
  }, []);

  return (
    <div className="group relative inline-flex">
      {/* Magnetic glow effect */}
      <div 
        className="absolute inset-0 rounded-xl blur-xl bg-liquid-accent transition-all duration-300 ease-out"
        style={{
          opacity: isHovered ? glowIntensity : 0,
          transform: `translate(${position.x * 0.5}px, ${position.y * 0.5}px) scale(${isHovered ? 1.1 : 1})`
        }}
      />
      
      {/* Main button with magnetic effect */}
      <GlassButton
        ref={buttonRef}
        className="relative z-10 transition-transform duration-200 ease-out"
        style={{
          transform: `translate(${position.x}px, ${position.y}px)`
        }}
        onMouseMove={handleMouseMove}
        onMouseEnter={() => setIsHovered(true)}
        onMouseLeave={handleMouseLeave}
        {...props}
      >
        {children}
      </GlassButton>
    </div>
  );
}
```

### Magnetic Card Hover

```tsx
export function MagneticCard({ children, className = "", ...props }) {
  const cardRef = useRef<HTMLDivElement>(null);
  const [rotateX, setRotateX] = useState(0);
  const [rotateY, setRotateY] = useState(0);

  const handleMouseMove = (e: React.MouseEvent) => {
    if (!cardRef.current) return;
    
    const rect = cardRef.current.getBoundingClientRect();
    const centerX = rect.left + rect.width / 2;
    const centerY = rect.top + rect.height / 2;
    
    const rotateXValue = (e.clientY - centerY) / 10;
    const rotateYValue = (centerX - e.clientX) / 10;
    
    setRotateX(rotateXValue);
    setRotateY(rotateYValue);
  };

  return (
    <GlassCard
      ref={cardRef}
      className={`group cursor-pointer transition-all duration-300 ${className}`}
      style={{
        transform: `perspective(1000px) rotateX(${rotateX}deg) rotateY(${rotateY}deg)`,
        transformStyle: 'preserve-3d'
      }}
      onMouseMove={handleMouseMove}
      onMouseLeave={() => {
        setRotateX(0);
        setRotateY(0);
      }}
      {...props}
    >
      <div className="relative z-10" style={{ transform: 'translateZ(50px)' }}>
        {children}
      </div>
      
      {/* Depth shadow */}
      <div className="absolute inset-0 rounded-xl bg-gradient-to-br from-liquid-accent/10 to-liquid-accent/5 opacity-0 group-hover:opacity-100 transition-opacity duration-300" />
    </GlassCard>
  );
}
```

### Multi-Step Form with Progress

```tsx
import { useState } from 'react';
import { GlassFormField, GlassInput, GlassButton, GlassProgress, GlassCard } from '@liquidify/react';

export function MultiStepForm() {
  const [currentStep, setCurrentStep] = useState(0);
  const [formData, setFormData] = useState({
    name: '',
    email: '',
    company: '',
    preferences: {}
  });

  const steps = [
    { id: 'personal', title: 'Personal Information', icon: 'üë§' },
    { id: 'contact', title: 'Contact Details', icon: 'üìß' },
    { id: 'preferences', title: 'Preferences', icon: '‚öôÔ∏è' },
    { id: 'review', title: 'Review & Submit', icon: '‚úÖ' }
  ];

  const progress = ((currentStep + 1) / steps.length) * 100;

  return (
    <GlassCard variant="elevated" className="max-w-2xl mx-auto p-8">
      {/* Progress header */}
      <div className="mb-8">
        <div className="flex items-center justify-between mb-4">
          {steps.map((step, index) => (
            <div
              key={step.id}
              className={`flex items-center space-x-2 ${
                index <= currentStep ? 'text-liquid-accent' : 'text-liquid-tertiary'
              }`}
            >
              <div className={`w-8 h-8 rounded-full flex items-center justify-center text-sm font-medium transition-colors ${
                index <= currentStep 
                  ? 'bg-liquid-accent text-white' 
                  : 'bg-liquid-surface-secondary text-liquid-tertiary'
              }`}>
                {index < currentStep ? '‚úì' : step.icon}
              </div>
              <span className="hidden sm:block text-sm font-medium">
                {step.title}
              </span>
            </div>
          ))}
        </div>
        
        <GlassProgress value={progress} className="h-2" />
      </div>

      {/* Step content */}
      <form className="space-y-6">
        {currentStep === 0 && (
          <div className="space-y-4">
            <h2 className="text-xl font-semibold text-liquid-primary mb-6">
              Personal Information
            </h2>
            <GlassFormField label="Full Name" required>
              <GlassInput
                value={formData.name}
                onChange={(e) => setFormData(prev => ({ ...prev, name: e.target.value }))}
                placeholder="Enter your full name"
              />
            </GlassFormField>
          </div>
        )}

        {currentStep === 1 && (
          <div className="space-y-4">
            <h2 className="text-xl font-semibold text-liquid-primary mb-6">
              Contact Details
            </h2>
            <GlassFormField label="Email Address" required>
              <GlassInput
                type="email"
                value={formData.email}
                onChange={(e) => setFormData(prev => ({ ...prev, email: e.target.value }))}
                placeholder="your@email.com"
              />
            </GlassFormField>
          </div>
        )}

        {/* Navigation */}
        <div className="flex justify-between pt-6">
          <GlassButton
            variant="secondary"
            onClick={() => setCurrentStep(prev => Math.max(0, prev - 1))}
            disabled={currentStep === 0}
          >
            Previous
          </GlassButton>
          
          <GlassButton
            onClick={() => setCurrentStep(prev => Math.min(steps.length - 1, prev + 1))}
            disabled={currentStep === steps.length - 1}
          >
            {currentStep === steps.length - 1 ? 'Submit' : 'Next'}
          </GlassButton>
        </div>
      </form>
    </GlassCard>
  );
}
```

### Real-time Validation Form

```tsx
import { useState, useEffect } from 'react';
import { GlassFormField, GlassInput, GlassButton, GlassIcon } from '@liquidify/react';

export function ValidatedEmailForm() {
  const [email, setEmail] = useState("");
  const [password, setPassword] = useState("");
  const [errors, setErrors] = useState<Record<string, string>>({});
  const [isValid, setIsValid] = useState(false);

  const validateEmail = (value: string) => {
    const emailRegex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
    return emailRegex.test(value);
  };

  const validatePassword = (value: string) => {
    return value.length >= 8 && /(?=.*[a-z])(?=.*[A-Z])(?=.*\d)/.test(value);
  };

  useEffect(() => {
    const newErrors: Record<string, string> = {};
    
    if (email && !validateEmail(email)) {
      newErrors.email = "Please enter a valid email address";
    }
    
    if (password && !validatePassword(password)) {
      newErrors.password = "Password must be at least 8 characters with uppercase, lowercase, and number";
    }
    
    setErrors(newErrors);
    setIsValid(email && password && Object.keys(newErrors).length === 0);
  }, [email, password]);

  return (
    <form className="space-y-6 max-w-md mx-auto">
      <GlassFormField
        label="Email Address"
        error={errors.email}
        success={email && !errors.email ? "Valid email format" : undefined}
        required
      >
        <GlassInput
          type="email"
          value={email}
          onChange={(e) => setEmail(e.target.value)}
          placeholder="your@company.com"
          className={`${
            email && !errors.email 
              ? 'border-liquid-success focus:ring-liquid-success' 
              : errors.email 
              ? 'border-liquid-danger focus:ring-liquid-danger' 
              : ''
          }`}
          rightIcon={
            email && !errors.email ? (
              <GlassIcon name="check-circle" className="text-liquid-success" />
            ) : errors.email ? (
              <GlassIcon name="exclamation-circle" className="text-liquid-danger" />
            ) : null
          }
        />
      </GlassFormField>

      <GlassFormField
        label="Password"
        error={errors.password}
        success={password && !errors.password ? "Strong password" : undefined}
        description="Minimum 8 characters with uppercase, lowercase, and number"
        required
      >
        <GlassInput
          type="password"
          value={password}
          onChange={(e) => setPassword(e.target.value)}
          placeholder="Enter secure password"
          className={`${
            password && !errors.password 
              ? 'border-liquid-success focus:ring-liquid-success' 
              : errors.password 
              ? 'border-liquid-danger focus:ring-liquid-danger' 
              : ''
          }`}
        />
      </GlassFormField>

      <GlassButton
        type="submit"
        disabled={!isValid}
        className="w-full"
        variant={isValid ? "primary" : "secondary"}
      >
        {isValid ? "Create Account" : "Complete Form"}
      </GlassButton>
    </form>
  );
}
```

### Adaptive Navigation Pattern

```tsx
import { useState, useEffect } from 'react';
import { GlassButton, GlassSheet, GlassIcon } from '@liquidify/react';

export function AdaptiveNavigation() {
  const [isMobile, setIsMobile] = useState(false);
  const [isMenuOpen, setIsMenuOpen] = useState(false);

  useEffect(() => {
    const checkScreenSize = () => {
      setIsMobile(window.innerWidth < 768);
    };
    
    checkScreenSize();
    window.addEventListener('resize', checkScreenSize);
    return () => window.removeEventListener('resize', checkScreenSize);
  }, []);

  const navigationItems = [
    { label: 'Overview', href: '#overview', icon: 'home' },
    { label: 'Components', href: '#components', icon: 'squares-2x2' },
    { label: 'Patterns', href: '#patterns', icon: 'puzzle-piece' },
    { label: 'Resources', href: '#resources', icon: 'book-open' }
  ];

  if (isMobile) {
    return (
      <>
        <div className="flex items-center justify-between p-4 bg-liquid-surface/80 backdrop-blur-xl border-b border-liquid-border">
          <h1 className="font-semibold text-liquid-primary">Liquidify</h1>
          <GlassButton
            variant="ghost"
            size="sm"
            onClick={() => setIsMenuOpen(true)}
          >
            <GlassIcon name="bars-3" />
          </GlassButton>
        </div>

        <GlassSheet open={isMenuOpen} onOpenChange={setIsMenuOpen}>
          <div className="p-6 space-y-4">
            {navigationItems.map((item) => (
              <GlassButton
                key={item.label}
                variant="ghost"
                className="w-full justify-start"
                onClick={() => setIsMenuOpen(false)}
              >
                <GlassIcon name={item.icon} className="mr-3" />
                {item.label}
              </GlassButton>
            ))}
          </div>
        </GlassSheet>
      </>
    );
  }

  return (
    <nav className="flex items-center space-x-1 p-4 bg-liquid-surface/80 backdrop-blur-xl border-b border-liquid-border">
      <h1 className="font-semibold text-liquid-primary mr-8">Liquidify</h1>
      {navigationItems.map((item) => (
        <GlassButton key={item.label} variant="ghost" size="sm">
          <GlassIcon name={item.icon} className="mr-2" />
          {item.label}
        </GlassButton>
      ))}
    </nav>
  );
}
```

### Toast Queue System

```tsx
import { useState, useCallback } from 'react';
import { GlassToast, GlassButton, GlassIcon } from '@liquidify/react';

interface Toast {
  id: string;
  message: string;
  type: 'success' | 'error' | 'warning' | 'info';
  duration?: number;
}

export function ToastSystem() {
  const [toasts, setToasts] = useState<Toast[]>([]);

  const addToast = useCallback((toast: Omit<Toast, 'id'>) => {
    const id = Math.random().toString(36).substr(2, 9);
    const newToast = { ...toast, id };
    
    setToasts(current => [...current, newToast]);
    
    setTimeout(() => {
      setToasts(current => current.filter(t => t.id !== id));
    }, toast.duration || 5000);
  }, []);

  const removeToast = useCallback((id: string) => {
    setToasts(current => current.filter(t => t.id !== id));
  }, []);

  return (
    <div className="space-y-4">
      {/* Demo buttons */}
      <div className="flex gap-2 flex-wrap">
        <GlassButton
          variant="ghost"
          onClick={() => addToast({ message: 'Changes saved successfully!', type: 'success' })}
        >
          Success Toast
        </GlassButton>
        <GlassButton
          variant="ghost"
          onClick={() => addToast({ message: 'Failed to connect to server', type: 'error' })}
        >
          Error Toast
        </GlassButton>
        <GlassButton
          variant="ghost"
          onClick={() => addToast({ message: 'Please review your settings', type: 'warning' })}
        >
          Warning Toast
        </GlassButton>
      </div>

      {/* Toast container */}
      <div className="fixed bottom-4 right-4 space-y-2 z-50 max-w-sm">
        {toasts.map((toast) => (
          <GlassToast
            key={toast.id}
            variant={toast.type}
            className="animate-in slide-in-from-right-full duration-300"
            onClose={() => removeToast(toast.id)}
          >
            <div className="flex items-center space-x-3">
              <GlassIcon 
                name={
                  toast.type === 'success' ? 'check-circle' :
                  toast.type === 'error' ? 'x-circle' :
                  toast.type === 'warning' ? 'exclamation-triangle' :
                  'information-circle'
                } 
                className={`w-5 h-5 ${
                  toast.type === 'success' ? 'text-liquid-success' :
                  toast.type === 'error' ? 'text-liquid-danger' :
                  toast.type === 'warning' ? 'text-liquid-warning' :
                  'text-liquid-info'
                }`}
              />
              <span className="flex-1 text-sm">{toast.message}</span>
            </div>
          </GlassToast>
        ))}
      </div>
    </div>
  );
}
```
