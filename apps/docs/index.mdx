## 🚀 Quick Start

<Steps>
  <Step title="Install Liquidify">
    Add LiqUIdify to your React project with your preferred package manager
    ```bash
    npm install liquidify
    ```
  </Step>
  <Step title="Setup Theme Provider">
    Wrap your app with the ThemeProvider for automatic theme switching
    ```tsx
    import "liquidify/styles";

    export default function App() {
      return (
        <ThemeProvider>
          <YourApp />
        </ThemeProvider>
      );
    }
    ```

  </Step>
  <Step title="Use Components">
    Start building with beautiful, accessible components
    ```tsx
    
    function MyComponent() {
      return (
        <Card className="liquid-glass">
          <h2>Welcome to LiqUIdify</h2>
          <Button variant="primary">Get Started</Button>
        </Card>
      );
    }
    ```

  </Step>
</Steps>
