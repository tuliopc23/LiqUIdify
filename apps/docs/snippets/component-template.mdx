```jsx React Code
import { GlassButton } from 'liquidify';

function MyComponent() {
  return (
    <GlassButton 
      variant="primary"
      onClick={() => console.log('Clicked!')}
    >
      Click Me
    </GlassButton>
  );
}
```

```tsx TypeScript
import { GlassButton } from 'liquidify';
import type { GlassButtonProps } from 'liquidify';

interface MyComponentProps {
  label: string;
  onAction: () => void;
}

function MyComponent({ label, onAction }: MyComponentProps) {
  const buttonProps: GlassButtonProps = {
    variant: "primary",
    onClick: onAction
  };

  return (
    <GlassButton {...buttonProps}>
      {label}
    </GlassButton>
  );
}
```

