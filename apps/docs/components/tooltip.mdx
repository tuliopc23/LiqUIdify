---
title: Tooltip
description: Contextual information on hover/focus with liquid glass styling.
---

import { GlassButton, GlassTooltip } from "@liquidify/react";

### Example

```tsx
export default function Demo() {
  return (
    <div className="flex gap-4 items-center justify-center p-8">
      <GlassTooltip content="This is a helpful tooltip">
        <GlassButton>Hover me</GlassButton>
      </GlassTooltip>

      <GlassTooltip content="Positioned on the right" position="right">
        <GlassButton variant="ghost">Right tooltip</GlassButton>
      </GlassTooltip>
    </div>
  );
}
```

Props

- content: ReactNode, position?: top | bottom | left | right, delay?: number
