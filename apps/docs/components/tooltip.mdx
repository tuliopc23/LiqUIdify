# Tooltip

The Tooltip component displays helpful information when users hover over or focus on an element. It's perfect for providing context, explanations, or additional details without cluttering the interface.

## Basic Usage

```tsx
import { Tooltip, Button } from '@liquidify/react';

function BasicTooltip() {
  return (
    <Tooltip content="This is a helpful tooltip">
      <Button>Hover me</Button>
    </Tooltip>
  );
}
```

## Disabled State

Tooltips can be conditionally disabled:

```tsx
import { Tooltip, Button } from '@liquidify/react';
import { useState } from 'react';

function DisabledTooltip() {
  const [disabled, setDisabled] = useState(false);

  return (
    <div className="flex flex-col gap-4">
      <div className="flex items-center gap-2">
        <input
          type="checkbox"
          checked={disabled}
          onChange={(e) => setDisabled(e.target.checked)}
        />
        <label>Disable tooltip</label>
      </div>
      
      <Tooltip
        content="This tooltip can be disabled"
        isDisabled={disabled}
      >
        <Button>
          {disabled ? 'Tooltip Disabled' : 'Tooltip Enabled'}
        </Button>
      </Tooltip>
    </div>
  );
}
```

## API Reference

### Tooltip Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `content` | `ReactNode` | - | Tooltip content |
| `placement` | `Placement` | `'top'` | Tooltip position |
| `color` | `'default' \| 'primary' \| 'secondary' \| 'success' \| 'warning' \| 'danger'` | `'default'` | Tooltip color |
| `size` | `'sm' \| 'md' \| 'lg'` | `'md'` | Tooltip size |
| `trigger` | `'hover' \| 'focus' \| 'click' \| 'manual'` | `'hover'` | Trigger behavior |
| `delay` | `number` | `400` | Show delay in ms |
| `closeDelay` | `number` | `0` | Hide delay in ms |
| `isOpen` | `boolean` | - | Controlled open state |
| `defaultOpen` | `boolean` | `false` | Default open state |
| `isDisabled` | `boolean` | `false` | Disabled state |
| `showArrow` | `boolean` | `true` | Show arrow pointer |
| `arrowSize` | `number` | `8` | Arrow size in pixels |
| `offset` | `number` | `8` | Distance from trigger |
| `interactive` | `boolean` | `false` | Allow interaction with tooltip |
| `className` | `string` | - | Additional CSS classes |
| `onOpenChange` | `(isOpen: boolean) => void` | - | Open state change handler |
| `children` | `ReactNode` | - | Trigger element |

## Best Practices

### 1. Keep Content Concise

Tooltips should provide brief, helpful information:

```tsx
// Good
<Tooltip content="Save your changes">
  <Button>Save</Button>
</Tooltip>

// Avoid
<Tooltip content="This button will save all of your changes to the database and update the user interface to reflect the new state">
  <Button>Save</Button>
</Tooltip>
```

### 2. Use Appropriate Triggers

Choose the right trigger for your use case:

```tsx
// Hover for informational tooltips
<Tooltip content="Additional info" trigger="hover">
  <InfoIcon />
</Tooltip>

// Click for interactive tooltips
<Tooltip content={<ActionMenu />} trigger="click" interactive>
  <MenuButton />
</Tooltip>

// Focus for form field help
<Tooltip content="Enter your email" trigger="focus">
  <Input type="email" />
</Tooltip>
```

### 3. Consider Mobile Users

Hover doesn't work on mobile, so provide alternatives:

```tsx
<Tooltip
  content="Help text"
  trigger={isMobile ? 'click' : 'hover'}
>
  <HelpIcon />
</Tooltip>
```

### Status Indicator Tooltips

```tsx
function StatusIndicators() {
  const servers = [
    { name: 'Web Server', status: 'online', uptime: '99.9%' },
    { name: 'Database', status: 'warning', uptime: '98.5%' },
    { name: 'API Gateway', status: 'offline', uptime: '0%' }
  ];

  const getStatusColor = (status: string) => {
    switch (status) {
      case 'online': return 'success';
      case 'warning': return 'warning';
      case 'offline': return 'danger';
      default: return 'default';
    }
  };

  return (
    <div className="space-y-3">
      {servers.map((server) => (
        <div key={server.name} className="flex items-center gap-3">
          <Tooltip
            content={
              <div>
                <div className="font-semibold">{server.name}</div>
                <div className="text-sm">Status: {server.status}</div>
                <div className="text-sm">Uptime: {server.uptime}</div>
              </div>
            }
            color={getStatusColor(server.status)}
          >
            <div className={`w-3 h-3 rounded-full ${
              server.status === 'online' ? 'bg-green-500' :
              server.status === 'warning' ? 'bg-yellow-500' :
              'bg-red-500'
            }`} />
          </Tooltip>
          <span>{server.name}</span>
        </div>
      ))}
    </div>
  );
}
```
