# GlassButton

A liquid glass button component with smooth animations and Apple-inspired aesthetics.

## Overview

The GlassButton component provides a beautiful glassmorphism effect with interactive hover states and smooth animations. Perfect for modern web applications that need an elegant, translucent aesthetic.

```tsx TypeScript
import { GlassButton } from "liquidify";
import type { GlassButtonProps } from "liquidify";

interface ButtonDemoProps {
  label: string;
  action: () => void;
}

function ButtonDemo({ label, action }: ButtonDemoProps) {
  const handleClick = () => {
    action();
    console.log("Glass button clicked!");
  };

  return (
    <GlassButton variant="primary" size="lg" onClick={handleClick}>
      {label}
    </GlassButton>
  );
}
```

## Usage Examples

### Basic Usage

The simplest way to use the GlassButton:

```jsx
import { GlassButton } from "liquidify";

<GlassButton>Click me!</GlassButton>;
```

### With Loading State

```jsx
import { GlassButton } from "liquidify";

<GlassButton loading={isLoading} onClick={handleSubmit}>
  {isLoading ? "Submitting..." : "Submit Form"}
</GlassButton>;
```

### Custom Styling

```jsx
import { GlassButton } from "liquidify";

<GlassButton
  className="custom-glass-button"
  style={{
    background: "rgba(139, 92, 246, 0.3)",
    borderColor: "rgba(139, 92, 246, 0.5)",
  }}
>
  Custom Style
</GlassButton>;
```

## Best Practices

<Callout type="info">
  Use glass buttons on backgrounds with sufficient contrast to maintain
  readability and visual hierarchy.
</Callout>

<Callout type="warning">
  Avoid using multiple glass elements in close proximity as it can reduce visual
  clarity.
</Callout>

<Callout type="tip">
  Combine with subtle animations and hover effects for enhanced user experience.
</Callout>
