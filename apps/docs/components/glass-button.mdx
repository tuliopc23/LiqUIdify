import { GlassButton } from "@liquidify/react";

# Glass Button

High quality button primitive built on the Liquid Glass system with adaptive variant blending, accessible focus management, optional loading states, and icon composition.

```tsx
<div className="flex flex-wrap gap-3">
  <GlassButton variant="primary">Primary</GlassButton>
  <GlassButton variant="secondary">Secondary</GlassButton>
  <GlassButton variant="ghost">Ghost</GlassButton>
  <GlassButton variant="destructive">Destructive</GlassButton>
</div>
```

## Installation

<Tip>Install the Liquidify package to get access to all glass components.</Tip>

```bash
npm install @liquidify/react
```

## Basic Usage

```tsx
import { GlassButton } from "@liquidify/react";

export function MyComponent() {
  return (
    <GlassButton variant="primary" onClick={() => console.log("Clicked!")}>
      Click me
    </GlassButton>
  );
}
```

## Loading States

<Info>
  Glass buttons include built-in loading states with smooth spinner animations.
</Info>

```tsx
<div className="flex gap-3">
  <GlassButton variant="primary" loading loadingText="Saving...">
    Save Changes
  </GlassButton>
  <GlassButton variant="ghost" loading>
    Loading
  </GlassButton>
</div>
```

## API Reference

<ResponseField
  name="variant"
  type="'primary' | 'secondary' | 'ghost' | 'destructive'"
  default="'primary'"
>
  Button style variant
</ResponseField>

<ResponseField name="loading" type="boolean" default="false">
  Show loading spinner
</ResponseField>

<ResponseField name="loadingText" type="string" default="'Loading...'">
  Text to show when loading
</ResponseField>

<ResponseField name="iconOnly" type="boolean" default="false">
  Icon-only button styling
</ResponseField>

<ResponseField name="leftIcon" type="ReactNode" default="undefined">
  Icon to show on the left
</ResponseField>

<Check>
  **Accessibility:** Glass buttons include proper focus indicators, keyboard
  navigation, and ARIA attributes for screen readers.
</Check>
