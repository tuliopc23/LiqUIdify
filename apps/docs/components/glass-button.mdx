## <span className="sf-icon-text"><span className="sf-icon sf-gear"></span>Installation</span>

<Tip>Install the Liquidify package to get access to all glass components.</Tip>

```bash
npm install @liquidify/react
```

## <span className="sf-icon-text"><span className="sf-icon sf-chevron-right"></span>Basic Usage</span>

```tsx
import { GlassButton } from "@liquidify/react";

function MyComponent() {
  return (
    <GlassButton variant="primary" onClick={() => console.log("Clicked!")}>
      Click me
    </GlassButton>
  );
}
```

## <span className="sf-icon-text"><span className="sf-icon sf-bookmark"></span>API Reference</span>

<ResponseField
  name="variant"
  type="'primary' | 'ghost' | 'solid' | 'destructive'"
  default="'primary'"
>
  Button style variant
</ResponseField>

<ResponseField name="loading" type="boolean" default="false">
  Show loading spinner
</ResponseField>

<ResponseField name="loadingText" type="string" default="'Loading...'">
  Text to show when loading
</ResponseField>

<ResponseField name="iconOnly" type="boolean" default="false">
  Icon-only button styling
</ResponseField>

<ResponseField name="leftIcon" type="ReactNode" default="undefined">
  Icon to show on the left
</ResponseField>

<Check>
