## <div className="flex items-center gap-3 text-title-2 mb-4"><SFSymbol name="play-circle" size="xl" color="var(--ios-green)" />Basic Usage</div>

```tsx
import { GlassButton } from "@liquidify/react";

export function MyComponent() {
  return (
    <GlassButton variant="primary" onClick={() => console.log("Clicked!")}>
      <div className="flex items-center gap-2">
        <SFSymbol name="hand-tap" size="sm" color="white" />
        Click me
      </div>
    </GlassButton>
  );
}
```

## Loading States

<Info>
  Glass buttons include built-in loading states with smooth spinner animations.
</Info>

```tsx
<div className="flex gap-3">
  <GlassButton variant="primary" loading loadingText="Saving...">
    Save Changes
  </GlassButton>
  <GlassButton variant="ghost" loading>
    Loading
  </GlassButton>
</div>
```

## API Reference

<ResponseField
  name="variant"
  type="'primary' | 'secondary' | 'ghost' | 'destructive'"
  default="'primary'"
>
  Button style variant
</ResponseField>

<ResponseField name="loading" type="boolean" default="false">
  Show loading spinner
</ResponseField>

<ResponseField name="loadingText" type="string" default="'Loading...'">
  Text to show when loading
</ResponseField>

<ResponseField name="iconOnly" type="boolean" default="false">
  Icon-only button styling
</ResponseField>

<ResponseField name="leftIcon" type="ReactNode" default="undefined">
  Icon to show on the left
</ResponseField>
