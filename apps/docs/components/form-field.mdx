### Example

```tsx
export default function Demo() {
  const [value, setValue] = useState("");
  const [email, setEmail] = useState("");
  const [password, setPassword] = useState("");

  return (
    <div className="space-y-6" style={{ maxWidth: 420 }}>
      {/* Basic form field */}
      <GlassFormField
        label="Full Name"
        required
        helperText="Enter your first and last name"
      >
        <GlassInput
          placeholder="John Doe"
          value={value}
          onChange={(e) => setValue(e.target.value)}
        />
      </GlassFormField>

      {/* Error state */}
      <GlassFormField
        label="Email Address"
        required
        error="Please enter a valid email address"
      >
        <GlassInput
          type="email"
          placeholder="john@example.com"
          value={email}
          onChange={(e) => setEmail(e.target.value)}
        />
      </GlassFormField>

      {/* Success state */}
      <GlassFormField label="Password" success="Password meets requirements">
        <GlassInput
          type="password"
          placeholder="••••••••"
          value={password}
          onChange={(e) => setPassword(e.target.value)}
        />
      </GlassFormField>

      {/* Inline variant */}
      <GlassFormField
        variant="inline"
        label="Newsletter"
        helperText="Receive updates"
      >
        <input type="checkbox" className="ml-auto" />
      </GlassFormField>
    </div>
  );
}
```

## Props

| Prop              | Type                                             | Default     | Description                                         |
| ----------------- | ------------------------------------------------ | ----------- | --------------------------------------------------- |
| `label`           | `string`                                         | -           | Field label text                                    |
| `required`        | `boolean`                                        | `false`     | Shows required asterisk                             |
| `error`           | `boolean \| string`                              | -           | Error state and message                             |
| `errorMessage`    | `string`                                         | -           | Custom error message                                |
| `success`         | `string`                                         | -           | Success message                                     |
| `warning`         | `string`                                         | -           | Warning message                                     |
| `helperText`      | `string`                                         | -           | Helper text below field                             |
| `variant`         | `'default' \| 'inline' \| 'card'`                | `'default'` | Visual variant                                      |
| `size`            | `'sm' \| 'md' \| 'lg'`                           | `'md'`      | Size variant                                        |
| `disabled`        | `boolean`                                        | `false`     | Disabled state                                      |
| `htmlFor`         | `string`                                         | -           | HTML for attribute (auto-generated if not provided) |
| `liveValidation`  | `boolean`                                        | `true`      | Enable live validation announcements                |
| `validationState` | `'default' \| 'error' \| 'success' \| 'warning'` | -           | Custom validation state                             |
| `focusOptions`    | `object`                                         | `{}`        | Focus management options                            |
| `children`        | `ReactNode`                                      | -           | Form field content                                  |

## Usage Patterns

### Basic Form

```tsx
function ContactForm() {
  const [formData, setFormData] = useState({
    name: "",
    email: "",
    message: "",
  });

  return (
    <form className="space-y-6">
      <GlassFormField label="Name" required>
        <GlassInput
          value={formData.name}
          onChange={(e) =>
            setFormData((prev) => ({ ...prev, name: e.target.value }))
          }
          placeholder="Your name"
        />
      </GlassFormField>

      <GlassFormField label="Email" required>
        <GlassInput
          type="email"
          value={formData.email}
          onChange={(e) =>
            setFormData((prev) => ({ ...prev, email: e.target.value }))
          }
          placeholder="your@email.com"
        />
      </GlassFormField>

      <GlassFormField label="Message">
        <textarea
          value={formData.message}
          onChange={(e) =>
            setFormData((prev) => ({ ...prev, message: e.target.value }))
          }
          placeholder="Your message"
          rows={4}
          className="w-full rounded-lg border border-white/20 bg-white/5 p-3 text-white backdrop-blur-md"
        />
      </GlassFormField>

      <GlassButton type="submit" variant="primary">
        Send Message
      </GlassButton>
    </form>
  );
}
```

### Validation with Error Handling

```tsx
function ValidatedForm() {
  const [email, setEmail] = useState("");
  const [errors, setErrors] = useState({});

  const validateEmail = (value) => {
    if (!value) return "Email is required";
    if (!/\S+@\S+\.\S+/.test(value)) return "Invalid email format";
    return null;
  };

  const handleEmailChange = (e: React.ChangeEvent<HTMLInputElement>) => {
    const value = e.target.value;
    setEmail(value);

    const error = validateEmail(value);
    setErrors((prev) => ({ ...prev, email: error }));
  };

  return (
    <GlassFormField
      label="Email Address"
      required
      error={errors.email}
      success={email && !errors.email ? "Valid email address" : undefined}
    >
      <GlassInput
        type="email"
        value={email}
        onChange={handleEmailChange}
        placeholder="Enter your email"
      />
    </GlassFormField>
  );
}
```

## Accessibility

### WCAG Compliance

- **Labels**: All form fields have proper labels with `htmlFor` attributes
- **Required Fields**: Required fields are marked with `aria-required` and visual asterisk
- **Error States**: Errors use `role="alert"` and `aria-live="polite"` for screen reader announcements
- **Descriptions**: Helper text is properly associated using `aria-describedby`
- **Focus Management**: Supports auto-focus and focus-on-error functionality

### Screen Reader Support

```tsx
// Automatic ARIA attributes
<GlassFormField
  label="Password"
  required
  error="Password must be at least 8 characters"
  helperText="Use a mix of letters, numbers, and symbols"
>
  <GlassInput type="password" />
</GlassFormField>

// Generates:
// <label for="field-id">Password *</label>
// <input
//   id="field-id"
//   aria-required="true"
//   aria-invalid="true"
//   aria-describedby="field-id-message"
// />
// <div id="field-id-message" role="alert" aria-live="polite">
//   Password must be at least 8 characters
// </div>
```

### Keyboard Navigation

- **Tab Order**: Proper tab sequence through form fields
- **Focus Indicators**: Clear focus rings on all interactive elements
- **Error Focus**: Automatic focus to fields with errors when `focusOnError` is enabled

## Apple HIG Guidelines

### Visual Design

- **Hierarchy**: Clear visual hierarchy with proper spacing and typography
- **Feedback**: Immediate visual feedback for validation states
- **Consistency**: Consistent spacing and alignment across form fields
- **Clarity**: Clear, concise labels and helper text

### Interaction Design

- **Progressive Disclosure**: Helper text appears when needed
- **Error Prevention**: Real-time validation helps prevent errors
- **Recovery**: Clear error messages help users recover from mistakes
- **Efficiency**: Inline variant for compact forms

### Motion and Animation

- **Subtle Transitions**: Smooth state changes with `motion-safe` support
- **Focus Animations**: Gentle focus transitions
- **State Changes**: Animated validation state transitions

## Best Practices

### Do's

- ✅ Use clear, descriptive labels
- ✅ Provide helpful error messages
- ✅ Mark required fields consistently
- ✅ Use appropriate input types
- ✅ Group related fields logically

### Don'ts

- ❌ Don't use placeholder text as labels
- ❌ Don't rely only on color for validation states
- ❌ Don't make error messages too technical
- ❌ Don't disable form submission on validation errors
- ❌ Don't use generic error messages

### Performance

- **Lazy Validation**: Validate on blur or submit, not on every keystroke
- **Debounced Updates**: Use debouncing for real-time validation
- **Minimal Re-renders**: Optimize state updates to prevent unnecessary renders
