import {
  DotsSpinner,
  GlassButton,
  GlassSpinner,
  PulseSpinner,
  RingSpinner,
  WaveSpinner,
} from "@liquidify/react";
import { useState } from "react";

### Example

```tsx
export default function Demo() {
  const [loading, setLoading] = useState(false);
  const simulateLoading = () => {
    setLoading(true);
    setTimeout(() => setLoading(false), 3000);
  };
  return (
    <div className="space-y-8 p-6">
      <div>
        <h3 className="text-liquid-text-inverse mb-4">
          Default Spinner Variants
        </h3>
        <div className="flex gap-6 items-center flex-wrap">
          <GlassSpinner variant="default" size="md" />
          <GlassSpinner variant="primary" size="md" />
          <GlassSpinner variant="success" size="md" />
          <GlassSpinner variant="warning" size="md" />
          <GlassSpinner variant="error" size="md" />
        </div>
      </div>
      <div>
        <h3 className="text-liquid-text-inverse mb-4">Sizes</h3>
        <div className="flex gap-6 items-center">
          <GlassSpinner size="xs" />
          <GlassSpinner size="sm" />
          <GlassSpinner size="md" />
          <GlassSpinner size="lg" />
          <GlassSpinner size="xl" />
          <GlassSpinner size="2xl" />
        </div>
      </div>
      <div>
        <h3 className="text-liquid-text-inverse mb-4">Alternative Spinners</h3>
        <div className="flex gap-8 items-center flex-wrap">
          <PulseSpinner size="md" />
          <DotsSpinner size="md" count={4} />
          <RingSpinner size="md" thickness={3} />
          <WaveSpinner bars={5} />
        </div>
      </div>
      <div>
        <h3 className="text-liquid-text-inverse mb-4">With Labels</h3>
        <div className="space-y-4">
          <GlassSpinner
            showLabel
            label="Loading data..."
            orientation="horizontal"
          />
          <GlassSpinner
            showLabel
            label="Processing..."
            orientation="vertical"
            variant="primary"
          />
        </div>
      </div>
      <div>
        <h3 className="text-liquid-text-inverse mb-4">Centered Overlay</h3>
        <div className="relative h-32 bg-liquid-bg/5 rounded-lg border border-liquid-highlight/10">
          <div className="p-4 text-liquid-text-inverse/70">
            Content behind spinner
          </div>
          {loading && <GlassSpinner centered showLabel label="Loading..." />}
        </div>
        <GlassButton onClick={simulateLoading} className="mt-4">
          Show Overlay Spinner
        </GlassButton>
      </div>
    </div>
  );
}
```

## Props

### GlassSpinner

| Prop          | Type                                                                                           | Default        | Description                   |
| ------------- | ---------------------------------------------------------------------------------------------- | -------------- | ----------------------------- |
| `variant`     | `'default' \| 'primary' \| 'secondary' \| 'success' \| 'warning' \| 'error' \| 'liquid-glass'` | `'default'`    | Color variant                 |
| `size`        | `'xs' \| 'sm' \| 'md' \| 'lg' \| 'xl' \| '2xl'`                                                | `'md'`         | Spinner size                  |
| `speed`       | `'slow' \| 'normal' \| 'fast'`                                                                 | `'normal'`     | Animation speed               |
| `orientation` | `'horizontal' \| 'vertical'`                                                                   | `'horizontal'` | Layout orientation with label |
| `label`       | `string`                                                                                       | `'Loading...'` | Accessible label text         |
| `showLabel`   | `boolean`                                                                                      | `false`        | Whether to show label text    |
| `centered`    | `boolean`                                                                                      | `false`        | Show as centered overlay      |
| `className`   | `string`                                                                                       | -              | Additional CSS classes        |

### Alternative Spinners

#### PulseSpinner

| Prop    | Type                   | Default   | Description |
| ------- | ---------------------- | --------- | ----------- |
| `size`  | `'sm' \| 'md' \| 'lg'` | `'md'`    | Dot size    |
| `color` | `string`               | `'white'` | Dot color   |

#### DotsSpinner

| Prop    | Type                   | Default | Description    |
| ------- | ---------------------- | ------- | -------------- |
| `size`  | `'sm' \| 'md' \| 'lg'` | `'md'`  | Dot size       |
| `count` | `number`               | `3`     | Number of dots |

#### RingSpinner

| Prop        | Type                   | Default | Description      |
| ----------- | ---------------------- | ------- | ---------------- |
| `size`      | `'sm' \| 'md' \| 'lg'` | `'md'`  | Ring size        |
| `thickness` | `number`               | `2`     | Border thickness |

#### WaveSpinner

| Prop   | Type     | Default | Description    |
| ------ | -------- | ------- | -------------- |
| `bars` | `number` | `5`     | Number of bars |

## Usage Patterns

### Button Loading State

```tsx
<GlassButton loading loadingText="Saving...">
  Save Changes
</GlassButton>
```

### Page Loading

```tsx
<GlassSpinner centered showLabel label="Loading page..." variant="primary" />
```

### Inline Loading

```tsx
<div className="flex items-center gap-2">
  <DotsSpinner size="sm" />
  <span>Processing...</span>
</div>
```

### Data Loading

```tsx
{
  isLoading ? (
    <GlassSpinner showLabel label="Fetching data..." />
  ) : (
    <DataTable data={data} />
  );
}
```

## Accessibility

- **ARIA Support**: Uses `role="status"` and `aria-label` for screen readers
- **Reduced Motion**: Respects `prefers-reduced-motion` settings
- **Screen Reader**: Announces loading state and progress
- **Focus Management**: Does not interfere with focus flow
- **Semantic HTML**: Uses appropriate semantic elements

## Apple HIG Guidelines

### Visual Design

- **Clarity**: Clear, recognizable loading indicators
- **Depth**: Subtle shadows and glass effects for visual hierarchy
- **Motion**: Smooth, continuous animations that feel natural
- **Color**: System-aware colors that adapt to light/dark modes

### Animation

- **Duration**: Appropriate timing for perceived performance
- **Easing**: Natural motion curves for smooth animation
- **Continuity**: Seamless transitions between states
- **Performance**: Optimized animations that don't impact performance

### User Experience

- **Feedback**: Immediate indication of loading state
- **Context**: Clear indication of what's being loaded
- **Cancellation**: Allow users to cancel long operations when possible
- **Progress**: Show progress when duration is known

### Best Practices

- Use appropriate spinner size for the context
- Provide meaningful loading messages
- Consider skeleton screens for better perceived performance
- Test animations on various devices and connection speeds
- Ensure spinners are visible against all background colors
