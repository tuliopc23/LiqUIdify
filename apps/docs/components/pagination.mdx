# Pagination

The Pagination component provides navigation controls for splitting large datasets across multiple pages. It includes customizable page controls, jump-to-page functionality, and responsive design.

## Basic Usage

```tsx
import { Pagination } from '@liquidify/react';

function BasicPagination() {
  const [currentPage, setCurrentPage] = useState(1);
  const totalPages = 10;

  return (
    <Pagination
      currentPage={currentPage}
      totalPages={totalPages}
      onPageChange={setCurrentPage}
    />
  );
}
```

## With Page Size Options

Allow users to change how many items are displayed per page:

```tsx
import { Pagination, Select, SelectItem } from '@liquidify/react';

function PaginationWithPageSize() {
  const [currentPage, setCurrentPage] = useState(1);
  const [pageSize, setPageSize] = useState(10);
  const totalItems = 250;
  const totalPages = Math.ceil(totalItems / pageSize);

  const handlePageSizeChange = (newPageSize: number) => {
    setPageSize(newPageSize);
    setCurrentPage(1); // Reset to first page
  };

  return (
    <div className="flex items-center justify-between">
      <div className="flex items-center gap-2">
        <span className="text-sm">Show</span>
        <Select
          value={pageSize.toString()}
          onValueChange={(value) => handlePageSizeChange(Number(value))}
        >
          <SelectItem value="5">5</SelectItem>
          <SelectItem value="10">10</SelectItem>
          <SelectItem value="25">25</SelectItem>
          <SelectItem value="50">50</SelectItem>
        </Select>
        <span className="text-sm">per page</span>
      </div>
      
      <Pagination
        currentPage={currentPage}
        totalPages={totalPages}
        onPageChange={setCurrentPage}
      />
    </div>
  );
}
```

## With Jump to Page

Add a quick jump-to-page input:

```tsx
import { Pagination, Input, Button } from '@liquidify/react';

function PaginationWithJump() {
  const [currentPage, setCurrentPage] = useState(1);
  const [jumpPage, setJumpPage] = useState('');
  const totalPages = 50;

  const handleJump = () => {
    const page = parseInt(jumpPage);
    if (page >= 1 && page <= totalPages) {
      setCurrentPage(page);
      setJumpPage('');
    }
  };

  const handleKeyPress = (e: React.KeyboardEvent) => {
    if (e.key === 'Enter') {
      handleJump();
    }
  };

  return (
    <div className="flex items-center gap-4">
      <Pagination
        currentPage={currentPage}
        totalPages={totalPages}
        onPageChange={setCurrentPage}
      />
      
      <div className="flex items-center gap-2">
        <span className="text-sm">Go to:</span>
        <Input
          type="number"
          value={jumpPage}
          onChange={(e) => setJumpPage(e.target.value)}
          onKeyPress={handleKeyPress}
          placeholder="Page"
          className="w-20"
          min={1}
          max={totalPages}
        />
        <Button size="sm" onClick={handleJump}>
          Go
        </Button>
      </div>
    </div>
  );
}
```

## Infinite Scroll Alternative

Use pagination as a "Load More" pattern:

```tsx
import { Pagination, Button } from '@liquidify/react';

function LoadMorePagination() {
  const [currentPage, setCurrentPage] = useState(1);
  const [items, setItems] = useState(generateInitialItems()); // Your initial data
  const hasMore = currentPage < 10; // Your logic to determine if more items exist

  const loadMore = () => {
    // Simulate loading more items
    const newItems = generateMoreItems(currentPage + 1);
    setItems(prev => [...prev, ...newItems]);
    setCurrentPage(prev => prev + 1);
  };

  return (
    <div>
      {/* Your items list */}
      <div className="grid gap-4 mb-6">
        {items.map((item) => (
          <ItemCard key={item.id} item={item} />
        ))}
      </div>
      
      {/* Load more controls */}
      <div className="flex flex-col items-center gap-4">
        {hasMore ? (
          <Button onClick={loadMore} variant="outline">
            Load More Items
          </Button>
        ) : (
          <p className="text-gray-500">No more items to load</p>
        )}
        
        <div className="text-sm text-gray-600">
          Loaded {items.length} items (Page {currentPage} of 10)
        </div>
      </div>
    </div>
  );
}
```

## Responsive Pagination

Adapts to different screen sizes:

```tsx
import { Pagination } from '@liquidify/react';

function ResponsivePagination() {
  const [currentPage, setCurrentPage] = useState(1);
  const totalPages = 25;

  return (
    <div className="flex flex-col sm:flex-row items-center justify-between gap-4">
      <div className="text-sm text-gray-600 order-2 sm:order-1">
        Page {currentPage} of {totalPages}
      </div>
      
      <Pagination
        currentPage={currentPage}
        totalPages={totalPages}
        onPageChange={setCurrentPage}
        className="order-1 sm:order-2"
        // Show fewer siblings on mobile
        siblingCount={window.innerWidth < 640 ? 0 : 1}
        showFirstLast={window.innerWidth >= 640}
      />
    </div>
  );
}
```

## API Reference

### Pagination Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `currentPage` | `number` | - | Current active page |
| `totalPages` | `number` | - | Total number of pages |
| `onPageChange` | `(page: number) => void` | - | Page change handler |
| `variant` | `'default' \| 'compact'` | `'default'` | Pagination variant |
| `size` | `'sm' \| 'md' \| 'lg'` | `'md'` | Button size |
| `siblingCount` | `number` | `1` | Number of siblings shown around current page |
| `boundaryCount` | `number` | `1` | Number of pages shown at start/end |
| `showFirstLast` | `boolean` | `true` | Show first/last page buttons |
| `showPrevNext` | `boolean` | `true` | Show previous/next buttons |
| `showEllipsis` | `boolean` | `true` | Show ellipsis for gaps |
| `disabled` | `boolean` | `false` | Disable all controls |
| `className` | `string` | - | Additional CSS classes |
| `pageButtonClassName` | `string` | - | Page button CSS classes |
| `activePageClassName` | `string` | - | Active page CSS classes |
| `disabledClassName` | `string` | - | Disabled button CSS classes |

## Best Practices

### 1. Provide Context

Always show users where they are and how much content exists:

```tsx
// Good
<div>
  <p>Showing 1-10 of 157 results</p>
  <Pagination currentPage={1} totalPages={16} onPageChange={setPage} />
</div>

// Better - with jump to page
<div>
  <p>Showing 1-10 of 157 results</p>
  <PaginationWithJump />
</div>
```

### 2. Handle Edge Cases

Ensure pagination works correctly at boundaries:

```tsx
function SafePagination({ currentPage, totalPages, onPageChange }) {
  const handlePageChange = (page: number) => {
    // Ensure page is within valid range
    const safePage = Math.max(1, Math.min(page, totalPages));
    onPageChange(safePage);
  };

  return (
    <Pagination
      currentPage={currentPage}
      totalPages={totalPages}
      onPageChange={handlePageChange}
    />
  );
}
```

### 3. Consider Performance

For large datasets, implement server-side pagination:

```tsx
function ServerPagination() {
  const [currentPage, setCurrentPage] = useState(1);
  const [pageSize] = useState(20);
  
  const { data, loading, totalPages } = useServerPagination({
    page: currentPage,
    pageSize,
    // other filters
  });

  return (
    <div>
      {loading ? <LoadingSpinner /> : <DataTable data={data} />}
      <Pagination
        currentPage={currentPage}
        totalPages={totalPages}
        onPageChange={setCurrentPage}
      />
    </div>
  );
}
```

### E-commerce Product Listing

```tsx
function ProductListing() {
  const [currentPage, setCurrentPage] = useState(1);
  const [pageSize, setPageSize] = useState(12);
  const [sortBy, setSortBy] = useState('name');
  
  const { products, totalCount, loading } = useProducts({
    page: currentPage,

  const totalPages = Math.ceil(totalCount / pageSize);

  return (

        <p className="text-gray-600">
```

### Search Results

```tsx
function SearchResults({ query }: { query: string }) {
  const [currentPage, setCurrentPage] = useState(1);
  const pageSize = 10;
  
  const { results, totalCount, loading } = useSearch({
    query,
    page: currentPage,
    pageSize
  });
  
  const totalPages = Math.ceil(totalCount / pageSize);
  const startResult = (currentPage - 1) * pageSize + 1;
  const endResult = Math.min(currentPage * pageSize, totalCount);

  if (loading) return <SearchSkeleton />;

  return (
    <div>
      <div className="mb-4">
        <h2 className="text-xl font-semibold mb-2">
          Search Results for "{query}"
        </h2>
        <p className="text-gray-600">
          About {totalCount.toLocaleString()} results ({results.searchTime}s)
        </p>
      </div>

      <div className="space-y-4 mb-8">
        {results.items.map((result) => (
          <SearchResultItem key={result.id} result={result} />
        ))}
      </div>

      {totalPages > 1 && (
        <div className="flex items-center justify-between">
          <p className="text-sm text-gray-600">
            Results {startResult}-{endResult} of {totalCount}
          </p>
          <Pagination
            currentPage={currentPage}
            totalPages={totalPages}
            onPageChange={setCurrentPage}
          />
        </div>
      )}
    </div>
  );
}
```
