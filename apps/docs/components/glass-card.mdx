# Glass Card

The `GlassCard` component is a versatile container that provides the perfect foundation for building elegant interfaces with glassmorphism effects. It supports compound components for structured content layouts.

## Import

<CodeGroup>

```tsx Component Import
import { GlassCard } from '@liquidify/components'
```

```tsx Bundle Import  
import { GlassCard } from '@liquidify/components/core'
```

```tsx Compound Components
import { 
  GlassCard,
  GlassCardHeader,
  GlassCardTitle,
  GlassCardDescription,
  GlassCardContent,
  GlassCardFooter 
} from '@liquidify/components'
```

</CodeGroup>

## Basic Usage

<CodeGroup>

```tsx Simple Card
<GlassCard className="p-6">
  <h3 className="text-lg font-semibold mb-2">Card Title</h3>
  <p className="text-gray-600">
    This is a simple glass card with basic content.
  </p>
</GlassCard>
```

```tsx With Variants
<div className="grid gap-4 md:grid-cols-2">
  <GlassCard variant="default" className="p-6">
    <h3>Default Card</h3>
    <p>Standard glass effect</p>
  </GlassCard>
  
  <GlassCard variant="elevated" className="p-6">
    <h3>Elevated Card</h3>
    <p>Enhanced elevation and depth</p>
  </GlassCard>
</div>
```

</CodeGroup>

## Props

<ResponseField name="variant" type="string" default="default">
  The visual style of the card
  - `"default"` - Standard glass effect
  - `"elevated"` - Enhanced depth and shadows
  - `"outline"` - Minimal style with border emphasis
  - `"ghost"` - Ultra-subtle glass effect
</ResponseField>

<ResponseField name="orientation" type="string" default="vertical">
  The layout orientation
  - `"vertical"` - Traditional card layout
  - `"horizontal"` - Side-by-side layout
</ResponseField>

<ResponseField name="hover" type="boolean" default="false">
  Enables enhanced hover effects
</ResponseField>

<ResponseField name="interactive" type="boolean" default="false">
  Makes the entire card clickable
</ResponseField>

<ResponseField name="selectable" type="boolean" default="false">
  Enables selection state styling
</ResponseField>

<ResponseField name="selected" type="boolean" default="false">
  Controls selection state (requires `selectable`)
</ResponseField>

<ResponseField name="bordered" type="boolean" default="false">
  Adds visible border to the card
</ResponseField>

<ResponseField name="elevation" type="string" default="none">
  Controls shadow depth
  - `"none"` - No shadow
  - `"sm"` - Small shadow
  - `"md"` - Medium shadow
  - `"lg"` - Large shadow
  - `"xl"` - Extra large shadow
</ResponseField>

<ResponseField name="glassEffect" type="object">
  Custom glass effect configuration
  ```tsx
  {
    intensity?: 'light' | 'medium' | 'strong' | 'ultra'
    blur?: boolean
    backdrop?: boolean
  }
  ```
</ResponseField>

<ResponseField name="asChild" type="boolean" default="false">
  When true, the component will render as its child element
</ResponseField>

## Events

<ResponseField name="onCardClick" type="(event: MouseEvent) => void">
  Called when the card is clicked (requires `interactive`)
</ResponseField>

<ResponseField name="onCardSelect" type="(selected: boolean) => void">
  Called when selection state changes (requires `selectable`)
</ResponseField>

<ResponseField name="onCardHover" type="(hovered: boolean) => void">
  Called when hover state changes
</ResponseField>

## Accessibility

LiqUIdify cards follow accessibility best practices:

- **Semantic Structure**: Uses appropriate HTML elements and ARIA roles
- **Keyboard Navigation**: Interactive cards support Tab and Enter keys
- **Screen Reader Support**: Proper labeling and state announcements
- **Focus Management**: Clear focus indicators for interactive states
- **Selection States**: Properly communicated via `aria-selected`

### Accessibility Example

```tsx
<GlassCard
  interactive
  selectable
  selected={isSelected}
  onCardClick={handleCardClick}
  onCardSelect={setIsSelected}
  aria-label="Product card for Wireless Headphones"
  aria-describedby="card-description"
  role="button"
  tabIndex={0}
>
  <GlassCardHeader>
    <GlassCardTitle id="card-title">
      Wireless Headphones  
    </GlassCardTitle>
    <GlassCardDescription id="card-description">
      Premium noise-cancelling headphones with 30-hour battery life
    </GlassCardDescription>
  </GlassCardHeader>
  
  <GlassCardContent>
    {/* Card content */}
  </GlassCardContent>
</GlassCard>
```

## Theming

Customize card appearance using CSS variables:

```css
.custom-glass-card {
  --glass-card-bg: rgba(255, 255, 255, 0.1);
  --glass-card-border: rgba(255, 255, 255, 0.2);
  --glass-card-shadow: 0 8px 32px rgba(0, 0, 0, 0.12);
  --glass-card-radius: 16px;
  
  /* Selected state */
  --glass-card-selected-bg: rgba(59, 130, 246, 0.1);
  --glass-card-selected-border: rgba(59, 130, 246, 0.3);
  
  /* Hover state */
  --glass-card-hover-bg: rgba(255, 255, 255, 0.15);
  --glass-card-hover-border: rgba(255, 255, 255, 0.3);
}
```

## Performance Notes

- Cards are optimized with `React.memo` for efficient re-renders
- Glass effects are GPU-accelerated using CSS transforms
- Intersection Observer used for scroll-based animations
- Event handlers are bound efficiently to minimize memory usage
- Large lists should use virtualization for optimal performance
