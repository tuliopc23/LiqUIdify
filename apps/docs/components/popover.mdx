# Popover

The Popover component displays rich content in a floating container that appears above other page content. Unlike tooltips, popovers can contain interactive elements and more complex layouts.

## Basic Usage

```tsx
import { Popover, PopoverTrigger, PopoverContent, Button } from '@liquidify/react';

function BasicPopover() {
  return (
    <Popover>
      <PopoverTrigger>
        <Button>Open Popover</Button>
      </PopoverTrigger>
      <PopoverContent>
        <div className="p-4">
          <h3 className="font-semibold mb-2">Popover Title</h3>
          <p className="text-sm text-gray-600">
            This is the popover content. It can contain any React elements.
          </p>
        </div>
      </PopoverContent>
    </Popover>
  );
}
```

## Placement

Control where the popover appears relative to the trigger:

```tsx
import { Popover, PopoverTrigger, PopoverContent, Button } from '@liquidify/react';

function PopoverPlacements() {
  const placements = [
    'top', 'top-start', 'top-end',
    'right', 'right-start', 'right-end',
    'bottom', 'bottom-start', 'bottom-end',
    'left', 'left-start', 'left-end'
  ];

  return (
    <div className="grid grid-cols-3 gap-4 p-8">
      {placements.map((placement) => (
        <Popover key={placement}>
          <PopoverTrigger>
            <Button variant="outline" size="sm">
              {placement}
            </Button>
          </PopoverTrigger>
          <PopoverContent placement={placement}>
            <div className="p-3">
              <p className="text-sm">Popover at {placement}</p>
            </div>
          </PopoverContent>
        </Popover>
      ))}
    </div>
  );
}
```

## Trigger Types

```tsx

```

## Form Popover

Use popovers for inline forms and inputs:

```tsx
import { 
  Popover, 
  PopoverTrigger, 
  PopoverContent, 
  Button,
  Input,
  Label,
  Textarea
} from '@liquidify/react';
import { Plus } from 'lucide-react';

function FormPopover() {
  const [isOpen, setIsOpen] = useState(false);

  const handleSubmit = (e: React.FormEvent) => {
    e.preventDefault();
    // Handle form submission
    setIsOpen(false);
  };

  return (
    <Popover isOpen={isOpen} onOpenChange={setIsOpen}>
      <PopoverTrigger>
        <Button>
          <Plus size={16} className="mr-2" />
          Add Comment
        </Button>
      </PopoverTrigger>
      <PopoverContent className="w-80">
        <form onSubmit={handleSubmit} className="p-4 space-y-4">
          <div>
            <Label htmlFor="title">Title</Label>
            <Input
              id="title"
              placeholder="Enter title"
              className="mt-1"
            />
          </div>
          <div>
            <Label htmlFor="comment">Comment</Label>
            <Textarea
              id="comment"
              placeholder="Enter your comment"
              rows={3}
              className="mt-1"
            />
          </div>
          <div className="flex justify-end gap-2">
            <Button
              type="button"
              variant="outline"
              size="sm"
              onClick={() => setIsOpen(false)}
            >
              Cancel
            </Button>
            <Button type="submit" size="sm">
              Submit
            </Button>
          </div>
        </form>
      </PopoverContent>
    </Popover>
  );
}
```

## Controlled Popover

Manage popover state programmatically:

```tsx
import { Popover, PopoverTrigger, PopoverContent, Button } from '@liquidify/react';
import { useState } from 'react';

function ControlledPopover() {
  const [isOpen, setIsOpen] = useState(false);

  return (
    <div className="flex flex-col gap-4">
      <div className="flex gap-2">
        <Button onClick={() => setIsOpen(true)}>Open Popover</Button>
        <Button onClick={() => setIsOpen(false)} variant="outline">
          Close Popover
        </Button>
        <Button onClick={() => setIsOpen(!isOpen)} variant="ghost">
          Toggle Popover
        </Button>
      </div>

      <Popover isOpen={isOpen} onOpenChange={setIsOpen}>
        <PopoverTrigger>
          <Button variant="outline">
            Controlled Trigger (State: {isOpen ? 'Open' : 'Closed'})
          </Button>
        </PopoverTrigger>
        <PopoverContent>
          <div className="p-4">
            <h3 className="font-semibold mb-2">Controlled Popover</h3>
            <p className="text-sm text-gray-600 mb-3">
              This popover's state is controlled externally.
            </p>
            <Button size="sm" onClick={() => setIsOpen(false)}>
              Close from Inside
            </Button>
          </div>
        </PopoverContent>
      </Popover>
    </div>
  );
}
```

## Nested Popovers

Popovers can be nested within each other:

```tsx
import { Popover, PopoverTrigger, PopoverContent, Button } from '@liquidify/react';

function NestedPopovers() {
  return (
    <Popover>
      <PopoverTrigger>
        <Button>Open Parent</Button>

      <PopoverContent>
        <div className="p-4">
          <h3 className="font-semibold mb-3">Parent Popover</h3>
          <p className="text-sm text-gray-600 mb-3">
            This popover contains another popover.
          </p>
          
          <Popover>
            <PopoverTrigger>
              <Button size="sm">Open Child</Button>
            </PopoverTrigger>
            <PopoverContent placement="right">

              </div>
            </PopoverContent>
          </Popover>
        </div>
      </PopoverContent>

  );
}
```

## API Reference

### Popover Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `isOpen` | `boolean` | - | Controlled open state |
| `defaultOpen` | `boolean` | `false` | Default open state |
| `onOpenChange` | `(isOpen: boolean) => void` | - | Open state change handler |
| `trigger` | `'click' \| 'hover' \| 'focus'` | `'click'` | Trigger behavior |
| `placement` | `Placement` | `'bottom'` | Popover position |
| `offset` | `number` | `8` | Distance from trigger |
| `crossOffset` | `number` | `0` | Cross-axis offset |
| `shouldFlip` | `boolean` | `true` | Flip when out of bounds |
| `shouldUpdatePosition` | `boolean` | `true` | Update position on scroll |
| `isKeyboardDismissDisabled` | `boolean` | `false` | Disable Escape key dismiss |
| `isNonModal` | `boolean` | `false` | Non-modal behavior |
| `isDismissable` | `boolean` | `true` | Allow outside click dismiss |
| `children` | `ReactNode` | - | Popover content |

### PopoverTrigger Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `children` | `ReactNode` | - | Trigger element |

### PopoverContent Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `showArrow` | `boolean` | `false` | Show arrow pointer |
| `arrowSize` | `number` | `12` | Arrow size in pixels |
| `className` | `string` | - | Additional CSS classes |
| `children` | `ReactNode` | - | Popover content |

## Best Practices

### 1. Use for Rich Content

Popovers are ideal for complex content that doesn't fit in tooltips:

```tsx
// Good - Rich interactive content
<Popover>
  <PopoverTrigger><Button>User Menu</Button></PopoverTrigger>
  <PopoverContent>
    <UserProfileCard />
    <ActionButtons />
  </PopoverContent>
</Popover>

// Avoid - Simple text (use Tooltip instead)
<Popover>
  <PopoverTrigger><Button>Help</Button></PopoverTrigger>
  <PopoverContent>
    <p>Simple help text</p>
  </PopoverContent>
</Popover>
```

### 2. Provide Clear Dismissal

Make it easy for users to close popovers:

```tsx
<Popover>
  <PopoverTrigger><Button>Open</Button></PopoverTrigger>
  <PopoverContent>
    <div className="p-4">
      <div className="flex justify-between items-center mb-3">
        <h3>Title</h3>
        <Button size="sm" variant="ghost" onClick={close}>
          <X size={16} />
        </Button>
      </div>
      <p>Content</p>
    </div>
  </PopoverContent>
</Popover>
```

### 3. Consider Mobile Experience

Ensure popovers work well on mobile devices:

```tsx
<Popover
  trigger={isMobile ? 'click' : 'hover'}
  placement={isMobile ? 'bottom' : 'right'}
>
  <PopoverTrigger><Button>Menu</Button></PopoverTrigger>
  <PopoverContent className={isMobile ? 'w-screen max-w-sm' : 'w-64'}>
    <MenuContent />
  </PopoverContent>
</Popover>
```

## Examples

### Color Picker Popover

```tsx
function ColorPickerPopover() {
  const [selectedColor, setSelectedColor] = useState('#3b82f6');
  const colors = [
    '#ef4444', '#f97316', '#f59e0b', '#eab308',
    '#84cc16', '#22c55e', '#10b981', '#14b8a6',
    '#06b6d4', '#0ea5e9', '#3b82f6', '#6366f1',
    '#8b5cf6', '#a855f7', '#d946ef', '#ec4899'
  ];

  return (
    <Popover>
      <PopoverTrigger>
        <Button
          variant="outline"
          className="w-20 h-10"
          style={{ backgroundColor: selectedColor }}
        >
          <span className="sr-only">Select color</span>
        </Button>
      </PopoverTrigger>
      <PopoverContent className="w-64">
        <div className="p-4">
          <h3 className="font-semibold mb-3">Choose Color</h3>
          <div className="grid grid-cols-4 gap-2 mb-4">
            {colors.map((color) => (
              <button
                key={color}
                className={`w-12 h-12 rounded-lg border-2 ${
                  selectedColor === color ? 'border-gray-400' : 'border-gray-200'
                }`}
                style={{ backgroundColor: color }}
                onClick={() => setSelectedColor(color)}
              />
            ))}
          </div>
          <Input
            type="text"
            value={selectedColor}
            onChange={(e) => setSelectedColor(e.target.value)}
            placeholder="#000000"
          />
        </div>
      </PopoverContent>
    </Popover>
  );
}
```
