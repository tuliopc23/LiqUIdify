# GlassCheckbox

The `GlassCheckbox` component provides a beautiful checkbox input with glassmorphism effects, indeterminate state support, and full accessibility features.

## Import

```tsx
import { GlassCheckbox } from "@liquidify/components";
```

## Basic Usage

<CodeGroup>

```tsx Basic Checkbox
<GlassCheckbox>I agree to the terms and conditions</GlassCheckbox>
```

```tsx Controlled Checkbox
function ControlledExample() {
  const [checked, setChecked] = useState(false);

  return (
    <GlassCheckbox checked={checked} onCheckedChange={setChecked}>
      Subscribe to newsletter
    </GlassCheckbox>
  );
}
```

```tsx With Description
<GlassCheckbox

  description="Receive updates about new products and features"
>
  Marketing emails
```

## Component API

### GlassCheckbox

| Prop              | Type                         | Default | Description              |
| ----------------- | ---------------------------- | ------- | ------------------------ |
| `checked`         | `boolean`                    | -       | Controlled checked state |
| `defaultChecked`  | `boolean`                    | `false` | Default checked state    |
| `indeterminate`   | `boolean`                    | `false` | Indeterminate state      |
| `disabled`        | `boolean`                    | `false` | Disable checkbox         |
| `required`        | `boolean`                    | `false` | Required field           |
| `invalid`         | `boolean`                    | `false` | Invalid state            |
| `size`            | `'sm' \| 'md' \| 'lg'`       | `'md'`  | Checkbox size            |
| `onCheckedChange` | `(checked: boolean) => void` | -       | Change handler           |
| `description`     | `string`                     | -       | Help text                |
| `error`           | `string`                     | -       | Error message            |

## Examples

### Checkbox Group

```tsx
function CheckboxGroup() {
  const [selectedItems, setSelectedItems] = useState<string[]>([]);

  const handleItemChange = (value: string, checked: boolean) => {
    setSelectedItems((prev) =>
      checked ? [...prev, value] : prev.filter((item) => item !== value),
    );
  };

  const allSelected = selectedItems.length === 3;
  const noneSelected = selectedItems.length === 0;
  const someSelected = selectedItems.length > 0 && selectedItems.length < 3;

  return (
    <div className="space-y-4">
      <GlassCheckbox
        checked={allSelected}
        indeterminate={someSelected}
        onCheckedChange={(checked) => {
          setSelectedItems(checked ? ["item1", "item2", "item3"] : []);
        }}
      >
        Select All
      </GlassCheckbox>

      <div className="space-y-2 ml-6">
        <GlassCheckbox
          checked={selectedItems.includes("item1")}
          onCheckedChange={(checked) => handleItemChange("item1", checked)}
        >
          Option 1
        </GlassCheckbox>

        <GlassCheckbox
          checked={selectedItems.includes("item2")}
          onCheckedChange={(checked) => handleItemChange("item2", checked)}
        >
          Option 2
        </GlassCheckbox>

        <GlassCheckbox
          checked={selectedItems.includes("item3")}
          onCheckedChange={(checked) => handleItemChange("item3", checked)}
        >
          Option 3
        </GlassCheckbox>
      </div>
    </div>
  );
}
```

### Form Integration

```tsx
function FormExample() {
  const [formData, setFormData] = useState({
    terms: false,
    newsletter: false,
    notifications: true,
  });

  const [errors, setErrors] = useState<Record<string, string>>({});

  const handleSubmit = (e: React.FormEvent) => {
    e.preventDefault();

    const newErrors: Record<string, string> = {};

    if (!formData.terms) {
      newErrors.terms = "You must accept the terms and conditions";
    }

    setErrors(newErrors);

    if (Object.keys(newErrors).length === 0) {
      // Submit form
      console.log("Form submitted:", formData);
    }
  };

  return (
    <form onSubmit={handleSubmit} className="space-y-4">
      <GlassCheckbox
        checked={formData.terms}
        onCheckedChange={(checked) =>
          setFormData((prev) => ({ ...prev, terms: checked }))
        }
        required
        invalid={!!errors.terms}
        error={errors.terms}
      >
        I agree to the{" "}
        <GlassButton variant="link" size="sm">
          terms and conditions
        </GlassButton>
      </GlassCheckbox>

      <GlassCheckbox
        checked={formData.newsletter}
        onCheckedChange={(checked) =>
          setFormData((prev) => ({ ...prev, newsletter: checked }))
        }
        description="Receive updates about new products and features"
      >
        Subscribe to newsletter
      </GlassCheckbox>

      <GlassCheckbox
        checked={formData.notifications}
        onCheckedChange={(checked) =>
          setFormData((prev) => ({ ...prev, notifications: checked }))
        }
        description="Get notified about important account activities"
      >
        Email notifications
      </GlassCheckbox>

      <GlassButton type="submit" className="mt-6">
        Create Account
      </GlassButton>
    </form>
  );
}
```

### Checkbox Card

```tsx
function CheckboxCard({ checked, onCheckedChange, title, description, price }) {
  return (
    <GlassCard
      className={cn(
        "cursor-pointer transition-all",
        checked && "ring-2 ring-primary ring-opacity-50",
      )}
      onClick={() => onCheckedChange(!checked)}
      interactive
    >
      <GlassCard.Header>
        <div className="flex items-start justify-between">
          <div>
            <GlassCard.Title>{title}</GlassCard.Title>
            <GlassCard.Description>{description}</GlassCard.Description>
          </div>
          <GlassCheckbox
            checked={checked}
            onCheckedChange={onCheckedChange}
            onClick={(e) => e.stopPropagation()}
          />
        </div>
      </GlassCard.Header>
      <GlassCard.Footer>
        <div className="text-lg font-semibold">{price}</div>
      </GlassCard.Footer>
    </GlassCard>
  );
}

function PricingSelection() {
```

## Accessibility

The `GlassCheckbox` component follows WAI-ARIA guidelines:

- **Keyboard Navigation**: Space toggles the checkbox state
- **ARIA Attributes**: Proper `role`, `aria-checked`, and `aria-describedby`
- **Focus Management**: Clear focus indicators and proper tab order
- **Screen Readers**: Associated labels and descriptions are announced

### ARIA Implementation

```tsx
<GlassCheckbox
  id="newsletter"
  aria-describedby="newsletter-description"
  description="We'll send you updates once a week"
>
  Subscribe to newsletter
</GlassCheckbox>
```
