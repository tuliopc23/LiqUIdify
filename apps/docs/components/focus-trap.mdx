# Focus Trap

```tsx live
import { GlassFocusTrap, GlassButton } from "liquidify";
import { useState } from 'react';

export default function Demo(){
  const [open,setOpen]=useState(false);
  return (
    <div>
      <GlassButton onClick={()=>setOpen(o=>!o)}>{open? 'Close':'Open'} Panel</GlassButton>
      {open && (
        <GlassFocusTrap>
          <div className="p-4 bg-liquid-readable rounded-xl space-y-2">
            <input className="liquid-glass-input" placeholder="First" />
            <input className="liquid-glass-input" placeholder="Second" />
            <GlassButton onClick={()=>setOpen(false)}>Done</GlassButton>
          </div>
        </GlassFocusTrap>
      )}
    </div>
  );
}
```

## Notes

Focus is restored to trigger element when trap unmounts.
