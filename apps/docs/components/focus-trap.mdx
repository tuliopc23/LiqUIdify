import { GlassButton, GlassFocusTrap } from "@liquidify/react";
import { useState } from "react";

# Focus Trap

### Example

```tsx
export default function Demo() {
  const [open, setOpen] = useState(false);
  return (
    <div>
      <GlassButton onClick={() => setOpen((o) => !o)}>
        {open ? "Close" : "Open"} Panel
      </GlassButton>
      {open && (
        <GlassFocusTrap>
          <div className="p-4 bg-liquid-readable rounded-xl space-y-2">
            <input className="" placeholder="First" />
            <input className="" placeholder="Second" />
            <GlassButton onClick={() => setOpen(false)}>Done</GlassButton>
          </div>
        </GlassFocusTrap>
      )}
    </div>
  );
}
```

## Notes

Focus is restored to trigger element when trap unmounts.
