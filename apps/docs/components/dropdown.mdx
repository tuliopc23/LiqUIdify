# Dropdown

Menu-style selection component anchored to a trigger element.

## Installation

```bash
npm install liquidify
```

## Basic Usage

```tsx
import { GlassDropdown, GlassButton } from "liquidify";
import { useState } from "react";

export default function Demo() {
  const [value, setValue] = useState("");
  const items = [
    { label: "Profile", value: "profile" },
    { label: "Settings", value: "settings" },
    { separator: true },
    { label: "Sign out", value: "logout" },
  ];
  
  return (
    <div className="space-y-2">
      <GlassDropdown
        trigger={<GlassButton variant="ghost">Open menu</GlassButton>}
        items={items}
        onSelect={setValue}
      />
      <div className="text-sm">Selected: {value || "â€”"}</div>
    </div>
  );
}
```

## Props

| Prop | Type | Description |
|------|------|--------------|
| `trigger` | `ReactNode` | Element that triggers the dropdown |
| `items` | `DropdownItem[]` | Array of menu items |
| `onSelect` | `(value: string) => void` | Callback when item is selected |

## Item Structure

```typescript
type DropdownItem = {
  label: string;
  value: string;
  disabled?: boolean;
  icon?: ReactNode;
  separator?: boolean;
};
```
