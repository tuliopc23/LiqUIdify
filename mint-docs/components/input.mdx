# GlassInput

The `GlassInput` component provides a beautiful text input with glass effects, validation, and accessibility features.

## Import

```tsx
import { GlassInput } from "@liquidify/components";
```

## Basic Usage

```tsx
<GlassInput placeholder="Enter your name" label="Full Name" />
```

## With Validation

```tsx
<GlassInput
  label="Email Address"
  type="email"
  placeholder="you@example.com"
  error="Please enter a valid email"
  required
/>
```

## States

<CodeGroup>

```tsx Default
<GlassInput placeholder="Default input" />
```

```tsx With Label
<GlassInput label="Username" placeholder="Enter username" />
```

```tsx With Error
<GlassInput label="Password" type="password" error="Password is required" />
```

```tsx Disabled
<GlassInput placeholder="Disabled input" disabled />
```

</CodeGroup>

## API Reference

<ResponseField name="label" type="string">
  Label text for the input
</ResponseField>

<ResponseField name="placeholder" type="string">
  Placeholder text
</ResponseField>

<ResponseField name="type" type="string" default="text">
  Input type (text, email, password, etc.)
</ResponseField>

<ResponseField name="error" type="string">
  Error message to display
</ResponseField>

<ResponseField name="disabled" type="boolean" default="false">
  Whether the input is disabled
</ResponseField>

<ResponseField name="required" type="boolean" default="false">
  Whether the input is required
</ResponseField>
